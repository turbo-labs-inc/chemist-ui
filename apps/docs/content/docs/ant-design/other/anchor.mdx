# Anchor

Anchor links for navigating within a page.

## Basic Usage

```tsx
import { Anchor } from 'antd';

export default function AnchorExample() {
  return (
    <Anchor
      items={[
        { key: 'part-1', href: '#part-1', title: 'Part 1' },
        { key: 'part-2', href: '#part-2', title: 'Part 2' },
      ]}
    />
  );
}
```
