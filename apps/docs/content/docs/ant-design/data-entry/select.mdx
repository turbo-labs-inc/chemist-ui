# Select

Dropdown select component for choosing values.

## Basic Usage

```tsx
import { Select } from 'antd';

export default function SelectExample() {
  return (
    <Select
      style={{ width: 200 }}
      options={[
        { value: '1', label: 'Option 1' },
        { value: '2', label: 'Option 2' },
      ]}
    />
  );
}
```
