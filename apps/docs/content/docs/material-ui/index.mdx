# Material UI Components

## Installation

```bash
pnpm add @chemist-ui/core @chemist-ui/react @mui/material @emotion/react @emotion/styled
```

## Usage

```tsx
import { ChemistProvider, materialUiTheme } from '@chemist-ui/react';
import { Button } from '@mui/material';

function App() {
  return (
    <ChemistProvider designSystem="material-ui" theme={materialUiTheme}>
      <Button variant="contained">Themed Button</Button>
    </ChemistProvider>
  );
}
```
