# Button Group

Group buttons for related actions.

## Basic Usage

```tsx
import { ButtonGroup, Button } from '@mui/material';

export default function ButtonGroupExample() {
  return (
    <ButtonGroup variant="contained">
      <Button>One</Button>
      <Button>Two</Button>
      <Button>Three</Button>
    </ButtonGroup>
  );
}
```
