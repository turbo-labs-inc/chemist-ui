# Image

Image component with preview functionality.

## Basic Usage

```tsx
import { Image } from 'antd';

export default function ImageExample() {
  return (
    <Image
      width={200}
      src="https://zos.alipayobjects.com/rmsportal/jkjgkEfvpUPVyRjUImniVslZfWPnJuuZ.png"
    />
  );
}
```
