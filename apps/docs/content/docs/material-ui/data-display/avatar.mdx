# Avatar

User avatar component.

## Basic Usage

```tsx
import { Avatar, Stack } from '@mui/material';

export default function AvatarExample() {
  return (
    <Stack direction="row" spacing={2}>
      <Avatar>H</Avatar>
      <Avatar alt="User" src="/path/to/image.jpg" />
    </Stack>
  );
}
```
