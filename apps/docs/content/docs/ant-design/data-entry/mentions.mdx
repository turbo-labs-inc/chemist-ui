# Mentions

Input component for mentioning users or items.

## Basic Usage

```tsx
import { Mentions } from 'antd';

export default function MentionsExample() {
  return (
    <Mentions
      options={[
        { value: 'Alice', label: 'Alice' },
        { value: 'Bob', label: 'Bob' },
      ]}
    />
  );
}
```
