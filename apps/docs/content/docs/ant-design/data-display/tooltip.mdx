# Tooltip

Display helpful text on hover.

## Basic Usage

```tsx
import { Tooltip, Button } from 'antd';

export default function TooltipExample() {
  return (
    <Tooltip title="Tooltip text">
      <Button>Hover me</Button>
    </Tooltip>
  );
}
```
