# Carousel

Carousel component for cycling through content.

## Basic Usage

```tsx
import { Carousel } from 'antd';

export default function CarouselExample() {
  return (
    <Carousel autoplay>
      <div><h3>1</h3></div>
      <div><h3>2</h3></div>
      <div><h3>3</h3></div>
    </Carousel>
  );
}
```
