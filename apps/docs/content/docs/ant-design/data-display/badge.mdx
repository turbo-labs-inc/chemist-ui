# Badge

Display a count or status indicator.

## Basic Usage

```tsx
import { Badge, Avatar } from 'antd';

export default function BadgeExample() {
  return (
    <>
      <Badge count={5}>
        <Avatar shape="square" size="large" />
      </Badge>
      <Badge dot>
        <Avatar shape="square" size="large" />
      </Badge>
    </>
  );
}
```
