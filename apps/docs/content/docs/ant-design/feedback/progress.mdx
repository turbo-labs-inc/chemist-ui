# Progress

Progress indicator for operations.

## Basic Usage

```tsx
import { Progress } from 'antd';

export default function ProgressExample() {
  return (
    <>
      <Progress percent={30} />
      <Progress percent={50} status="active" />
      <Progress percent={70} status="exception" />
      <Progress percent={100} />
    </>
  );
}
```
