# Divider

Divider component separates content into clear groups.

## Installation

```bash
pnpm add @chemist-ui/core @chemist-ui/react antd
```

## Basic Usage

```tsx
import { Divider } from 'antd';

export default function DividerExample() {
  return (
    <>
      <p>Lorem ipsum dolor sit amet.</p>
      <Divider />
      <p>Lorem ipsum dolor sit amet.</p>
      <Divider dashed />
      <p>Lorem ipsum dolor sit amet.</p>
    </>
  );
}
```
