## Installation

```bash
pnpm add @chemist-ui/core @chemist-ui/react antd
```

## Usage

```tsx
import { ChemistProvider, antDesignTheme } from '@chemist-ui/react';

function App() {
  return (
    <ChemistProvider designSystem="ant-design" theme={antDesignTheme}>
      {/* Your Ant Design components */}
    </ChemistProvider>
  );
}
```
