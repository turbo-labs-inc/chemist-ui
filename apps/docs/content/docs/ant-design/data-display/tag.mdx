# Tag

Tag component for categorizing or labeling.

## Basic Usage

```tsx
import { Tag } from 'antd';

export default function TagExample() {
  return (
    <>
      <Tag>Tag 1</Tag>
      <Tag color="blue">Tag 2</Tag>
      <Tag color="green">Tag 3</Tag>
    </>
  );
}
```
